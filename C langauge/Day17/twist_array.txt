 #include <stdio.h>
int main(void)
{
int a[5] = {5, 1, 15, 20, 25};
int i, j, m;
i = ++a[1];
printf("%d\n",i);// 2 
// pre-increment 

j = a[1]++; // a[1] = 3 // post increment 
printf("%d\n",j); //2 
// post increment 

m = a[i++];// m = 15 
// post increment --> i --> 2++ --> 3  
printf("%d\n",m);


printf("%d, %d, %d", i, j, m);
return 0;
}
